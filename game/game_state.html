129    def player_point_cards(self, player: int) -> List[Card]:
130        """Get all point cards that count towards a player's score.
131
132        This includes:
133        - Point cards on the player's field that haven't been stolen
134        - Point cards on the opponent's field that have been stolen by the player
135
136        Args:
137            player (int): The player index (0 or 1).
138
139        Returns:
140            List[Card]: List of cards that count towards the player's score.
141        """
142        point_cards = []
143        player_field = self.fields[player]
144        for card in player_field:
145            if card.purpose == Purpose.POINTS and not card.is_stolen():
146                point_cards.append(card)
147        opponent = (player + 1) % len(self.hands)
148        for card in self.fields[opponent]:
149            if card.purpose == Purpose.POINTS and card.is_stolen():
150                point_cards.append(card)
151        return point_cards

209        kings = [card for card in self.fields[player] if card.rank == Rank.KING]
210        num_kings = len(kings)
211
212        if num_kings == 0:
213            return 21
214        elif num_kings == 1:
215            return 14
216        elif num_kings == 2:
217            return 10
218        elif num_kings == 3:
219            return 5
220        else:
221            return 0

260    def update_state(self, action: Action) -> Tuple[bool, bool, Optional[int]]:
261        """Update the game state based on an action.
262
263        This is the main method for executing game actions. It handles:
264        - Drawing cards
265        - Playing point cards
266        - Scuttling cards
267        - One-off effects
268        - Countering
269        - Effect resolution
270
271        Args:
272            action (Action): The action to execute.
273
274        Returns:
275            Tuple[bool, bool, Optional[int]]: A tuple containing:
276                - bool: Whether the turn is finished
277                - bool: Whether the game should stop
278                - Optional[int]: The winner's index if game is over, None otherwise
279        """
280        turn_finished = False
281        should_stop = False
282        winner = None
283
284        if action.action_type == ActionType.DRAW:
285            self.draw_card()
286            turn_finished = True
287            return turn_finished, should_stop, winner
288        elif action.action_type == ActionType.POINTS:
289            if action.card is not None:
290                won = self.play_points(action.card)
291                turn_finished = True
292                if won:
293                    should_stop = True
294                    winner = self.winner()
295                    return turn_finished, should_stop, winner
296            else:
297                # Handle error: POINTS action requires a card
298                log_print("Error: POINTS action called without a card.")
299                return True, True, None # Stop game on error
300        elif action.action_type == ActionType.SCUTTLE:
301            if action.card is not None and action.target is not None:
302                self.scuttle(action.card, action.target)
303                turn_finished = True
304                should_stop = False  # scuttle doesn't end the game
305                winner = self.winner()
306                return turn_finished, should_stop, winner
307            else:
308                # Handle error: SCUTTLE action requires card and target
309                log_print("Error: SCUTTLE action called without card or target.")
310                return True, True, None # Stop game on error
311        elif action.action_type == ActionType.ONE_OFF:
312            if action.card is not None:
313                # Normal one-off handling for all cards
314                turn_finished, played_by = self.play_one_off(
315                    self.turn, action.card, None, None
316                )
317                if turn_finished:
318                    winner = self.winner()
319                    should_stop = winner is not None
320                    return turn_finished, should_stop, winner
321                self.resolving_one_off = True
322                self.one_off_card_to_counter = action.card
323                return turn_finished, should_stop, winner
324            else:
325                # Handle error: ONE_OFF action requires a card
326                log_print("Error: ONE_OFF action called without a card.")
327                return True, True, None # Stop game on error
328        elif action.action_type == ActionType.COUNTER:
329            if action.card is not None and action.target is not None:
330                action.card.purpose = Purpose.COUNTER
331                if action.card.played_by is not None: # Check played_by before use
332                    self.current_action_player = action.card.played_by
333                turn_finished, played_by = self.play_one_off(
334                    player=self.turn,
335                    card=action.target, # Target is the card being countered
336                    countered_with=action.card, # Card is the Two used to counter
337                    last_resolved_by=None,
338                )
339                if turn_finished:
340                    winner = self.winner()
341                    should_stop = winner is not None
342                    return turn_finished, should_stop, winner
343            else:
344                # Handle error: COUNTER action requires card and target
345                log_print("Error: COUNTER action called without card or target.")
346                return True, True, None # Stop game on error
347        elif action.action_type == ActionType.RESOLVE:
348            if action.target is not None:
349                turn_finished, played_by = self.play_one_off(
350                    self.turn, action.target, None, action.played_by
351                )
352                if turn_finished:
353                    winner = self.winner()
354                    should_stop = winner is not None
355                    return turn_finished, should_stop, winner
356            else:
357                # Handle error: RESOLVE action requires a target
358                log_print("Error: RESOLVE action called without a target.")
359                return True, True, None # Stop game on error
360        elif action.action_type == ActionType.FACE_CARD:
361            if action.card is not None:
362                won = self.play_face_card(action.card, action.target) # Target can be None for King/Queen
363                turn_finished = True
364                if won:
365                    should_stop = True
366                    winner = self.turn
367                return turn_finished, should_stop, winner
368            else:
369                # Handle error: FACE_CARD action requires a card
370                log_print("Error: FACE_CARD action called without a card.")
371                return True, True, None # Stop game on error
372        elif action.action_type == ActionType.JACK:
373            if action.card is not None and action.target is not None:
374                # Check if opponent has a queen on their field
375                # implement play_face_card with optional target
376                won = self.play_face_card(action.card, action.target)
377                turn_finished = True
378                if won:
379                    should_stop = True
380                    winner = self.turn
381                return turn_finished, should_stop, winner
382            else:
383                # Handle error: JACK action requires card and target
384                log_print("Error: JACK action called without card or target.")
385                return True, True, None # Stop game on error
386
387        return turn_finished, should_stop, winner

420        # Validate scuttle conditions
421        if (
422            card.point_value() == target.point_value()
423            and card.suit_value() <= target.suit_value()
424        ):
425            raise Exception(
426                "Invalid scuttle: Cannot scuttle with a lower or equal suit of the same rank"

428
429        # scuttle a points card
430        card_player = card.played_by
431        if card_player is not None:
432            if card in self.hands[card_player]:
433                self.hands[card_player].remove(card)
434        card.clear_player_info()
435        self.discard_pile.append(card)
436        for attached_card in card.attachments:
437            attached_card.clear_player_info()
438            self.discard_pile.append(attached_card)

440        target_player = target.played_by

450    def play_one_off(
451        self,
452        player: int,
453        card: Card,
454        countered_with: Optional[Card] = None,
455        last_resolved_by: Optional[int] = None,
456    ) -> Tuple[bool, Optional[int]]:
457        """
458        Play a one-off card.
459
460        Args:
461            player: The player playing the card
462            card: The card being played
463            countered_with: The TWO being used to counter, if any
464            last_resolved_by: The player who last resolved (didn't counter), if any
465
466        Returns:
467            Tuple[bool, int | None]:
468                - Whether the turn is finished
469                - The player who played the last action (for tracking counter chain)
470        """
471        # Initial play requires additional input
472        self.last_action_played_by = player
473        if player == self.turn and countered_with is None and last_resolved_by is None:
474            # Initial play of one-off, waiting for counter/resolve
475            return False, None
476
477        # Handle counter with a Two
478        if countered_with is not None:
479            # Validate counter card
480            if countered_with.point_value() != 2:
481                raise Exception("Counter must be a 2")
482            if countered_with.purpose != Purpose.COUNTER:
483                raise Exception(
484                    f"Counter must be with a purpose of counter, instead got {countered_with.purpose}"
485                )
486            counter_player = countered_with.played_by
487            if counter_player is not None:
488                other_player = (counter_player + 1) % len(self.hands)
489                # check if other player has a queen on their field
490                other_player_field = self.fields[other_player]
491                queen_on_opponent_field = any(
492                    card.rank == Rank.QUEEN for card in other_player_field
493                )
494                if queen_on_opponent_field:
495                    raise Exception(
496                        "Cannot counter with a two if opponent has a queen on their field"
497                    )
498
499            # Move counter card to discard pile
500            played_by = countered_with.played_by
501            if played_by is not None and countered_with in self.hands[played_by]:
502                self.hands[played_by].remove(countered_with)
503                self.discard_pile.append(countered_with)
504                countered_with.clear_player_info()
505
506            # Move the countered card to discard pile if it's still in hand
507            if card in self.hands[self.turn]:
508                self.hands[self.turn].remove(card)
509            if card not in self.discard_pile:
510                self.discard_pile.append(card)
511                card.clear_player_info()
512
513            # Update last action for counter chain
514            self.last_action_played_by = played_by
515            return False, played_by
516
517        # Handle resolution (no counter)
518        if last_resolved_by is not None:
519            self.last_action_played_by = player
520
521            if last_resolved_by != self.turn:
522                # Opponent didn't counter, so one-off resolves
523                if card in self.hands[self.turn]:
524                    self.hands[self.turn].remove(card)
525                card.purpose = Purpose.ONE_OFF
526                self.apply_one_off_effect(card)
527                card.clear_player_info()
528                if card not in self.discard_pile:
529                    self.discard_pile.append(card)
530            else:
531                # Original player accepts counter
532                # One-off is countered, move to discard
533                if card in self.hands[self.turn]:
534                    self.hands[self.turn].remove(card)
535                if card not in self.discard_pile:
536                    self.discard_pile.append(card)
537                card.clear_player_info()
538
539            # Turn is finished after resolution
540            return True, None
541
542        return True, None

451    def play_one_off(
452        self,
453        player: int,
454        card: Card,
455        countered_with: Optional[Card] = None,
456        last_resolved_by: Optional[int] = None,
457    ) -> Tuple[bool, Optional[int]]:
458        """
459        Play a one-off card.
460
461        Args:
462            player: The player playing the card
463            card: The card being played
464            countered_with: The TWO being used to counter, if any
465            last_resolved_by: The player who last resolved (didn't counter), if any
466
467        Returns:
468            Tuple[bool, int | None]:
469                - Whether the turn is finished
470                - The player who played the last action (for tracking counter chain)
471        """
472        # Initial play requires additional input
473        self.last_action_played_by = player
474        if player == self.turn and countered_with is None and last_resolved_by is None:
475            # Initial play of one-off, waiting for counter/resolve
476            return False, None
477
478        # Handle counter with a Two
479        if countered_with is not None:
480            # Validate counter card
481            if countered_with.point_value() != 2:
482                raise Exception("Counter must be a 2")
483            if countered_with.purpose != Purpose.COUNTER:
484                raise Exception(
485                    f"Counter must be with a purpose of counter, instead got {countered_with.purpose}"
486                )
487            counter_player = countered_with.played_by
488            if counter_player is not None:
489                other_player = (counter_player + 1) % len(self.hands)
490                # check if other player has a queen on their field
491                other_player_field = self.fields[other_player]
492                queen_on_opponent_field = any(
493                    card.rank == Rank.QUEEN for card in other_player_field
494                )
495                if queen_on_opponent_field:
496                    raise Exception(
497                        "Cannot counter with a two if opponent has a queen on their field"
498                    )
499
500            # Move counter card to discard pile
501            played_by = countered_with.played_by
502            if played_by is not None and countered_with in self.hands[played_by]:
503                self.hands[played_by].remove(countered_with)
504                self.discard_pile.append(countered_with)
505                countered_with.clear_player_info()
506
507            # Move the countered card to discard pile if it's still in hand
508            if card in self.hands[self.turn]:
509                self.hands[self.turn].remove(card)
510            if card not in self.discard_pile:
511                self.discard_pile.append(card)
512                card.clear_player_info()
513
514            # Update last action for counter chain
515            self.last_action_played_by = played_by
516            return False, played_by
517
518        # Handle resolution (no counter)
519        if last_resolved_by is not None:
520            self.last_action_played_by = player
521
522            if last_resolved_by != self.turn:
523                # Opponent didn't counter, so one-off resolves
524                if card in self.hands[self.turn]:
525                    self.hands[self.turn].remove(card)
526                card.purpose = Purpose.ONE_OFF
527                self.apply_one_off_effect(card)
528                card.clear_player_info()
529                if card not in self.discard_pile:
530                    self.discard_pile.append(card)
531            else:
532                # Original player accepts counter
533                # One-off is countered, move to discard
534                if card in self.hands[self.turn]:
535                    self.hands[self.turn].remove(card)
536                if card not in self.discard_pile:
537                    self.discard_pile.append(card)
538                card.clear_player_info()
539
540            # Turn is finished after resolution
541            return True, None
542
543        return True, None
